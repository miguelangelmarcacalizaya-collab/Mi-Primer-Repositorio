import java.util.Scanner;

public class PilaDeStrings {
    private String[] pila;
    private int tope;
    private int tama√±o;

    // Constructor
    public PilaDeStrings(int tama√±o) {
        this.tama√±o = tama√±o;
        pila = new String[tama√±o];
        tope = -1;
    }

    // Insertar un elemento
    public void push(String dato) {
        if (isFull()) {
            System.out.println("‚ùå La pila est√° llena.");
        } else {
            pila[++tope] = dato;
            System.out.println("‚úîÔ∏è Se agreg√≥: " + dato);
        }
    }

    // Eliminar un elemento
    public String pop() {
        if (isEmpty()) {
            System.out.println("‚ö†Ô∏è La pila est√° vac√≠a.");
            return null;
        } else {
            String dato = pila[tope--];
            System.out.println("üóëÔ∏è Se elimin√≥: " + dato);
            return dato;
        }
    }

    // Ver elemento superior
    public String peek() {
        if (isEmpty()) {
            System.out.println("‚ö†Ô∏è La pila est√° vac√≠a.");
            return null;
        } else {
            return pila[tope];
        }
    }

    // Mostrar todos los elementos
    public void mostrar() {
        if (isEmpty()) {
            System.out.println("‚ö†Ô∏è No hay elementos en la pila.");
        } else {
            System.out.println("\nüì¶ Contenido de la pila:");
            for (int i = tope; i >= 0; i--) {
                System.out.println("[" + i + "] " + pila[i]);
            }
        }
    }

    // Verificar si est√° vac√≠a
    public boolean isEmpty() {
        return tope == -1;
    }

    // Verificar si est√° llena
    public boolean isFull() {
        return tope == tama√±o - 1;
    }

    // Programa principal con men√∫
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        PilaDeStrings pila = new PilaDeStrings(5);
        int opcion;

        do {
            System.out.println("\n--- MEN√ö PILA EST√ÅTICA ---");
            System.out.println("1. Insertar palabra (push)");
            System.out.println("2. Eliminar palabra (pop)");
            System.out.println("3. Ver cima (peek)");
            System.out.println("4. Mostrar pila");
            System.out.println("5. Salir");
            System.out.print("Elige una opci√≥n: ");
            opcion = sc.nextInt();
            sc.nextLine(); // limpiar buffer

            switch (opcion) {
                case 1:
                    System.out.print("Escribe la palabra: ");
                    String palabra = sc.nextLine();
                    pila.push(palabra);
                    break;
                case 2:
                    pila.pop();
                    break;
                case 3:
                    String cima = pila.peek();
                    if (cima != null) System.out.println("üëÄ En la cima est√°: " + cima);
                    break;
                case 4:
                    pila.mostrar();
                    break;
                case 5:
                    System.out.println("üëã Saliendo del programa...");
                    break;
                default:
                    System.out.println("‚ùå Opci√≥n no v√°lida.");
            }
        } while (opcion != 5);
        sc.close();
    }
}
